## @repo/ui

Shared React components for use across applications.

**Location:** `packages/ui/`

### Package Configuration

```json packages/ui/package.json
{
  "name": "@repo/ui",
  "version": "0.0.0",
  "private": true,
  "exports": {
    "./*": "./src/*.tsx"
  },
  "scripts": {
    "lint": "eslint . --max-warnings 0",
    "generate:component": "turbo gen react-component",
    "check-types": "tsc --noEmit"
  },
  "devDependencies": {
    "@repo/eslint-config": "workspace:*",
    "@repo/typescript-config": "workspace:*",
    "@types/node": "^22.15.3",
    "@types/react": "19.1.0",
    "@types/react-dom": "19.1.1",
    "eslint": "^9.31.0",
    "typescript": "5.8.2"
  },
  "dependencies": {
    "react": "^19.1.0",
    "react-dom": "^19.1.0"
  }
}
```

### Components

#### Button Component

**Location:** `packages/ui/src/button.tsx`

```tsx packages/ui/src/button.tsx
"use client";

import { ReactNode } from "react";

interface ButtonProps {
  children: ReactNode;
  className?: string;
  appName: string;
}

export const Button = ({ children, className, appName }: ButtonProps) => {
  return (
    <button
      className={className}
      onClick={() => alert(`Hello from your ${appName} app!`)}
    >
      {children}
    </button>
  );
};
```

**Usage:**

```tsx
import { Button } from '@repo/ui/button'

<Button appName="web" className="btn-primary">
  Click Me
</Button>
```

#### Card Component

**Location:** `packages/ui/src/card.tsx`

```tsx packages/ui/src/card.tsx
import { type JSX } from "react";

export function Card({
  className,
  title,
  children,
  href,
}: {
  className?: string;
  title: string;
  children: React.ReactNode;
  href: string;
}): JSX.Element {
  return (
    <a
      className={className}
      href={`${href}?utm_source=create-turbo&utm_medium=basic&utm_campaign=create-turbo"`}
      rel="noopener noreferrer"
      target="_blank"
    >
      <h2>
        {title} <span>-&gt;</span>
      </h2>
      <p>{children}</p>
    </a>
  );
}
```

**Usage:**

```tsx
import { Card } from '@repo/ui/card'

<Card
  title="Documentation"
  href="https://turbo.build/repo"
  className="card"
>
  Learn about Turborepo
</Card>
```

#### Code Component

**Location:** `packages/ui/src/code.tsx`

```tsx packages/ui/src/code.tsx
import { type JSX } from "react";

export function Code({
  children,
  className,
}: {
  children: React.ReactNode;
  className?: string;
}): JSX.Element {
  return <code className={className}>{children}</code>;
}
```

**Usage:**

```tsx
import { Code } from '@repo/ui/code'

<p>
  Run <Code>pnpm dev</Code> to start development.
</p>
```

### Exports Pattern

The package uses wildcard exports:

```json
"exports": {
  "./*": "./src/*.tsx"
}
```

This allows importing any component:

```tsx
import { Button } from '@repo/ui/button'
import { Card } from '@repo/ui/card'
import { Code } from '@repo/ui/code'
```

### Component Generation

Create new components using Turbo generators:

```bash
cd packages/ui
pnpm generate:component
```

## @repo/constants

Shared constants and configuration values.

**Location:** `packages/constants/`

### Package Configuration

```json packages/constants/package.json
{
  "name": "@repo/constants",
  "type": "module",
  "exports": {
    ".": {
      "types": "./dist/constants.d.ts",
      "default": "./dist/constants.js"
    }
  },
  "scripts": {
    "build": "tsc",
    "dev": "tsc --watch"
  },
  "devDependencies": {
    "@repo/typescript-config": "workspace:*",
    "typescript": "^5.8.3"
  }
}
```

### Constants

**Location:** `packages/constants/src/constants.ts`

```typescript packages/constants/src/constants.ts
export const CONSTANT = 'Hello from your Hono API!'
```

### Usage

Import constants in any application:

<Tabs>
  <Tab title="API">
    ```typescript apps/api/src/index.ts
    import { CONSTANT } from '@repo/constants'
    
    export default {
      fetch: () => new Response(CONSTANT)
    }
    ```
  </Tab>
  
  <Tab title="Frontend">
    ```tsx
    import { CONSTANT } from '@repo/constants'
    
    export default function Page() {
      return <div>{CONSTANT}</div>
    }
    ```
  </Tab>
</Tabs>

### Building

The package must be built before use:

```bash
cd packages/constants
```

Or use watch mode during development:

```bash
pnpm dev
```

## Development Workflow

### Installing Shared Packages

Add a shared package to an app:

```bash
cd apps/web
pnpm add @repo/ui
```

Or manually in `package.json`:

```json
{
  "dependencies": {
    "@repo/ui": "workspace:*"
  }
}
```

### Building Packages

Build all packages:

```bash
pnpm build
```

Build specific package:

```bash
cd packages/constants
pnpm build
```

### Watch Mode

For active development:

```bash
cd packages/constants
pnpm dev
```

Changes are automatically recompiled.

## Adding a New Shared Package

<Steps>
  <Step title="Create Package Directory">
    ```bash
    mkdir -p packages/new-package/src
    cd packages/new-package
    ```
  </Step>
  
  <Step title="Create package.json">
    ```json
    {
      "name": "@repo/new-package",
      "version": "0.0.0",
      "private": true,
      "main": "./src/index.ts",
      "types": "./src/index.ts",
      "scripts": {
        "build": "tsc",
        "dev": "tsc --watch"
      }
    }
    ```
  </Step>
  
  <Step title="Create Source Files">
    ```bash
    echo 'export const MY_CONSTANT = "value"' > src/index.ts
    ```
  </Step>
  
  <Step title="Add TypeScript Config">
    ```json tsconfig.json
    {
      "extends": "@repo/typescript-config/base.json",
      "compilerOptions": {
        "outDir": "./dist"
      },
      "include": ["src"]
    }
    ```
  </Step>
  
  <Step title="Build and Use">
    ```bash
    pnpm build
    
    # In another package
    pnpm add @repo/new-package
    ```
  </Step>
</Steps>

## Package Dependencies

Shared packages can depend on each other:

```json packages/ui/package.json
{
  "dependencies": {
    "@repo/constants": "workspace:*"
  }
}
```

Turborepo handles the build order automatically.
